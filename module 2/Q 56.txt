import java.util.HashSet;

public class Hasset 
{
    public static void main(String[] args) 
    {
       
        HashSet<String> s = new HashSet<>();

        
        s.add("shdb");
        s.add("nxbh");
        s.add("jx");

      
        String ne = "mo";
        s.add(ne);

       
        System.out.println(" after appen :");
        for (String el : s) {
            System.out.println(el);
        }
    }
}
